SCRIPT AREA
START SCRIPT

DECLARE INT active = 1
DECLARE INT counter

REPEAT WHEN (active == 1)
START REPEAT
    
    PRINT: "Starting For Loop!"

    FOR (counter = 1, counter <= 3, counter = counter + 1)
    START FOR
        
        IF (counter == 2)
        START IF
            PRINT: "Counter is exactly 2!"
        END IF
        
        PRINT: counter
    END FOR

    PRINT: "Shutting down loop."
    active = 0 

END REPEAT

END SCRIPT
